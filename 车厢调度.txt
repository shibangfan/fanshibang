#include "pch.h"
#include<stdio.h>
#include<stdlib.h>

int final;
int total = 0;

typedef struct
{
	int *base;
	int *top;
	int size;
}sqs;

int Inits(sqs *s)
{
	s->base = (int*)malloc(final * sizeof(int));
	if (!s->base) exit(0);
	s->top = s->base;
	s->size = final;
}

void Push(sqs *s, int e)
{
	*(s->top)++ = e;
}

int Pop(sqs *s)
{
	if (s->top == s->base)
	{
		return 0;
	}
	else
	{
		return *(--(s->top));
	}
}

int Empty(sqs *s)
{
	if (s->top == s->base)
	{
		return 1;
	}
	else
	{
		return 0;
	}
}

int Fulls(sqs *s)
{
	if (s->top - s->base == final)
	{
		return 1;
	}
	else
	{
		return 0;
	}
}

void Print(sqs s)
{
	int *p;
	p = s.base;
	printf("\t[%d]:", total);
	for (; p != s.top; )
	{
		printf("%d", *p++);
	}
	printf("\n");
}

void search(sqs *input, sqs *temp, sqs *output)
{
	if (!Empty(input))
	{
		Push(temp, Pop(input));
		search(input, temp, output);
		Push(input, Pop(temp));
	}
	if (!Empty(temp))
	{
		Push(output, Pop(temp));
		search(input, temp, output);
		Push(temp, Pop(output));
	}
	if (Fulls(output))
	{
		total++;
		Print(*output);
	} 
}

void main()
{
	sqs input, temp, output;
	int i;
	printf("请输入车厢长度:\n");
	scanf_s("%d", &final);
	printf("车厢序列为:\n");
	Inits(&input);
	Inits(&temp);
	Inits(&output);
	for (i = final; i >= 1; i--)
	{
		Push(&input, i);
	}
	search(&input, &temp, &output);
}
